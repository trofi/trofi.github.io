122  * encoding with regard to current policy.

132 Range * Enc::encodeRange(uint32_t l, uint32_t h) const
133 {

135  {
136  return NULL;
137  }
138

140  switch (type_)

144  break;
145  case EBCDIC:

152  r = Range::add (r, Range::sym (ec));

157  case UTF16:
158  case UTF32:

162  {
