72 {
73  if(!eof)
74  {
75  /* Do not get rid of anything when rFlag is active. Otherwise

80  if (diff > 0)
81  {
82  const size_t move = static_cast<size_t> (top - tok);
83  memmove (bot, tok, move);
84  tok -= diff;
85  ptr -= diff;
86  cur -= diff;
87  pos -= diff;
88  lim -= diff;
89  ctx -= diff;
90  }
91  }

93  if (BSIZE > need)
94  {
95  need = BSIZE;
96  }
97  if (static_cast<uint32_t> (top - lim) < need)
98  {

100  char * buf = new char[copy + need];
101  if (!buf)

104  }
105  memcpy (buf, bot, copy);
106  tok = &buf[tok - bot];
107  ptr = &buf[ptr - bot];
108  cur = &buf[cur - bot];
109  pos = &buf[pos - bot];
110  lim = &buf[lim - bot];
111  top = &lim[need];
112  ctx = &buf[ctx - bot];
